#Planet Killed
planet_event = {
	id = sw22_on_action_events.3
	title = sw22_on_action_events.3.name
	desc = sw22_on_action_events.3.desc
	picture = GFX_evt_tomb_world
	location = ROOT
	#hide_window = yes
	is_triggered_only = yes

	trigger = {
		controller = {
			count_owned_army = {
				limit = { army_type = worm_army_rogue }
				count > 0
			}
		}
	}

	immediate = {
		change_pc = pc_nuked
		remove_all_buildings = yes
		set_planet_flag = nuked_planet_anomalies_disabled
		destroy_colony = yes
		add_planet_devastation = 100
	}	

	option = {
		name = sw22_on_action_events.3.a
		fromfromfrom = {
			add_opinion_modifier = { who = root.owner modifier = opinion_genocidal }
		}
	}

}
